Quick Start

1. Install
npm install clawsight
2. Add to your agent
const monitor = require('clawsight');
const watcher = monitor({
  server: 'https://app.clawsight.org',
  token:  'YOUR_API_KEY'
});

watcher.init({ id: 'agent-01', name: 'My Bot' });
watcher.log('Hello World', 'working');
watcher.metric('cost', 0.05);
